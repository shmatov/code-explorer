fn foo<T: Bar>(variable: T) {
    variable.bar();
    variable.barrr();
}

trait Bar {
    fn bar(&self);
    fn barrr(&self);
}

impl Bar for A<B> {
    fn bar(&self) { &self.field; }
    fn barrr(&self) { self.bar(); }
}

pub struct A<T> {
    field: T
}

enum B {
    BFirst,
    BSecond
}

mod foobarbaz;

fn main() {
    foo(A { field: B::BFirst });
    foobarbaz::hi(A { field: 100500i32 });
}
